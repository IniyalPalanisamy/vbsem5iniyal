Public Class Form1
Private Sub btn_clear_Click(sender As Object, e As EventArgs) Handles btn_clear.Click
Me.txt_allowance.Text = " "
Me.txt_empid.Text = " "
Me.txt_empname.Text = " "
Me.txt_epfamount.Text = " "
Me.txt_gpay.Text = " "
Me.txt_netpay.Text = " "
Me.txt_Bp.Text = " "
Me.txt_desig.Text = " "
End Sub
Private Sub btn_calculate_Click(sender As Object, e As EventArgs) Handles btn_calculate.Click
If chk_percentage.Checked = True Then
txt_epfamount.Text = Val(txt_Bp.Text) * 10 / 100
End If
If Val(txt_Bp.Text) > 50000 Then
txt_allowance.Text = 1500
Else
txt_allowance.Text = 500
End If
txt_gpay.Text = Val(txt_Bp.Text) + Val(txt_allowance.Text)
txt_netpay.Text = Val(txt_Bp.Text) - Val(txt_epfamount.Text) + Val(txt_allowance.Text)
End Sub
Private Sub btn_Exit_Click(sender As Object, e As EventArgs) Handles btn_Exit.Click
End
End Sub
End Class
